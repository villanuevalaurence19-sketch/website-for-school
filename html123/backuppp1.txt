    let currentIndex = 0;

    function updateCar(index) {
      const car = cars[index];
      const image = document.getElementById('car-image');
      image.classList.remove('zoomed');
      document.getElementById('car-title').textContent = car.title;
      document.getElementById('car-subtitle').textContent = car.subtitle;
      document.getElementById('car-price').textContent = car.price;
      image.src = car.image;
      document.getElementById('car-year').textContent = car.year;
      document.getElementById('car-mileage').textContent = car.mileage;
      document.getElementById('car-hp').textContent = car.hp;
    }

    function nextCar() {
      currentIndex = (currentIndex + 1) % cars.length;
      updateCar(currentIndex);
    }

    function prevCar() {
      currentIndex = (currentIndex - 1 + cars.length) % cars.length;
      updateCar(currentIndex);
    }

    function toggleZoom() {
      const image = document.getElementById('car-image');
      image.classList.toggle('zoomed');
    }

    updateCar(currentIndex);
